import matplotlib.pyplot as plt labels = [] quant = [] titulo = input().strip().title() def addDado(): label = 'label' quantia = 1 while(label!='0' and quantia!=0 and len(labels)<6): label = input().strip().title() if(label=='0'): break quantia = float(input().strip()) if(quantia>0): labels.append(label) quant.append(quantia) else: break addDado() fig, ax = plt.subplots(figsize=(4,3), dpi=155) ax.pie(quant, labels=labels, autopct='%.1f%%', startangle=90, shadow=False) ax.set_title(titulo) fig
